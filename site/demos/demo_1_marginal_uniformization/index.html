    Data¶

    seed = 123
n_samples = 10_000
a = 4

# initialize data distribution
data_dist = stats.gamma(a=a)

# get some samples
X = data_dist.rvs(size=(n_samples, 1), random_state=seed)

# plot distribution

    Fit Model to Distribution¶

    # initialize HistogramClass
nbins = 1_000 #int(np.sqrt(n_samples))
support_extension = 20
alpha = 1e-16

    nbins=nbins,
    support_extension=support_extension,
    alpha=alpha
)

# fit to data
